gitlab password:

psql password:
kubectl create secret generic psql-pword --from-literal=password="HelloWordl1234%"

tls:
Download pfx after certificate is created in AKV

openssl pkcs12 -in <filename>.pfx -nocerts -nodes -out key.pem
openssl pkcs12 -in <filename>.pfx -clcerts -nokeys -out cert.pem

kubectl create secret tls tls-secret --cert=cert.pem --key=key.pem

Refs https://support.microfocus.com/kb/doc.php?id=7004039
