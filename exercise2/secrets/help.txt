gitlab password:
kubectl create secret generic gitlab-pword --from-literal=password="HelloWorld1234%"

psql password:
kubectl create secret generic psql-pword --from-literal=password="HelloWorld1234%"

redis password:
kubectl create secret generic redis-pword --from-literal=password="HelloWorld1234%"

tls:
Download pfx after certificate is created in AKV

openssl pkcs12 -in secrets/.pfx -nocerts -nodes -out secrets/key.pem
openssl pkcs12 -in secrets/.pfx -clcerts -nokeys -out secrets/cert.pem

subchart tls
kubectl create secret tls tls-secret --cert=secrets/cert.pem --key=secrets/key.pem

CustomCa
kubectl create secret generic tls-ca --from-file=exercise-ca.crt=secrets/cert.pem

gitlab-runner
kubectl create secret generic gitlab-guillotinetech-cert --namespace default --from-file=gitlab-guillotinetech.crt=secrets/cert.pem

testing container
kubectl run ubuntu-test --image=ubuntu:latest --command -- sleep infinity
